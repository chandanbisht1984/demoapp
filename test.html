import { StringParameter } from 'aws-cdk-lib/aws-ssm'; import { Tags} from 'aws-cdk-lib' import { Construct } from 'constructs'; export const getParam = (scope: Construct, name: string) => { return StringParameter.valueForStringParameter(scope, name) } export const tagConstruct = (node: Construct, tags: Object)=>{ Object.entries(tags).forEach( ([key, value]) => { Tags.of(node).add(key,value); }); }
